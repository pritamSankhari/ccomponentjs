<script type="text/javascript" src="./CComponent.js"></script>

Script to add specific component

<script type="text/javascript">
	
	/* Component HTML */
	let product_html = function(){
 		return `
		<div>
			<product-name-label>Product Name:</product-name-label>
			<product-name>Green T-shirt</product-name>
		</div>
		<div>
			<product-price-label>Product Price:</product-price-label>
			<product-price>260/-</product-price>
		</div>
		`;
	}

	let product = new CComponent(document.querySelector(".root"),product_html,"product");
	product.add();

</script>

Script to add item

<div class="root"></div>
<button class="btn-add-item">+ Add Item</button>
<script type="text/javascript">

	/* Component HTML */
	let item_html = function(){
 		return `
		<div>
			<item-name-label>Item:</item-name-label>
			<item-name> Item_${this.id}</item-name>
		</div>
		`;
	}

	let item = new CComponent(document.querySelector(".root"),item_html,"item");
	
	document.querySelector(".btn-add-item").addEventListener("click",function(event){
		item.add();		
	})
	item.add();

</script>

Script to add item with delete button

Important: **Watch carefully the underlined code**

<div class="root"></div>
<button class="btn-add-item">+ Add Item</button>
<script type="text/javascript">

	/* Component HTML */
	let item_html = function(){
 		return `
		<div>
			<item-name-label>Item:</item-name-label>
			<item-name> Item_${this.id}</item-name>
		</div>
		<button item-index=${this.id} class="item-remove-btn">- Delete</button>
		`;
	}

	let item = new CComponent(document.querySelector(".root"),item_html,"item");
	
	item.addEventListenerForThis = function(element){

		this.addRemoveBtn(element,"item-remove-btn","item-index");
	}
	
	document.querySelector(".btn-add-item").addEventListener("click",function(event){
		item.add();		
	})
	item.add();

</script>

Note: this.addRemoveBtn should be called within addEventListenerForThis() method.  this.addRemoveBtn has three parameters - element, remove button classname and property name that holds unique id.
